CREATE DEFINER=`root`@`localhost` PROCEDURE `homeworkloop`()
begin
declare x int;
set x = 1;
while x <= 2 do
select homs_Agents_id,homs_price from homs where homs_price = (select max(homs_price) from homs where homs_Agents_id = x);
set x = x +1;
end while;
end

CREATE DEFINER=`root`@`localhost` PROCEDURE `homeworkloppinnerjoin`()
begin
declare x int;
set x = 1;
while x <= 2 do
select Agents_name,Agents_id 
from Agents inner join homs
on Agents.Agents_id = homs.homs_id
where homs.homs_price = (select max(homs_price) 
from homs where homs_Agents_id = x );

set x = x + 1;
end while;
end
